# Initial exploration
print("First 5 rows:\n", df.head())
print("\nMissing Values:\n", df.isnull().sum())
print("\nData Types:\n", df.dtypes)
print("\nUnique Values per Column:\n", df.nunique())

# Campaign acceptance rates
campaigns = ['Campaign 1', 'Campaign 2', 'Campaign 3', 'Campaign 4', 'Campaign 5']
acceptance_rates = [
    df['AcceptedCmp1'].mean() * 100,
    df['AcceptedCmp2'].mean() * 100,
    df['AcceptedCmp3'].mean() * 100,
    df['AcceptedCmp4'].mean() * 100,
    df['AcceptedCmp5'].mean() * 100
]
plt.bar(campaigns, acceptance_rates, color='blue')
plt.xlabel('Campaign')
plt.ylabel('Acceptance Rate (%)')
plt.title('Campaign Acceptance Rates')
plt.show()
print("Campaign Acceptance Rates (%):")
for camp, rate in zip(campaigns, acceptance_rates):
    print(f"{camp}: {rate:.2f}%")

# Complaint ratio
complaint_ratio = len(df[df['Complain'] == 1]) / len(df) * 100
print(f"\nComplaint Ratio: {complaint_ratio:.3f}%")

# Average recency
average_recency = df['Recency'].mean()
print(f"\nAverage Recency (Days): {average_recency:.2f}")

# Average purchases by channel
print("\nAverage Purchases by Channel:")
print(f"Discount: {df['NumDealsPurchases'].mean():.2f}")
print(f"Catalog: {df['NumCatalogPurchases'].mean():.2f}")
print(f"Web: {df['NumWebPurchases'].mean():.2f}")
print(f"Store: {df['NumStorePurchases'].mean():.2f}")

# Average spending by product
print("\nAverage Spending by Product:")
print(f"Fish: {df['MntFishProducts'].mean():.2f}")
print(f"Meat: {df['MntMeatProducts'].mean():.2f}")
print(f"Fruits: {df['MntFruits'].mean():.2f}")
print(f"Sweets: {df['MntSweetProducts'].mean():.2f}")
print(f"Gold: {df['MntGoldProds'].mean():.2f}")
print(f"Wines: {df['MntWines'].mean():.2f}")

# Correlation heatmap
plt.figure(figsize=(10, 8))
sns.heatmap(df.corr(), annot=False, cmap='coolwarm')
plt.title('Correlation Heatmap')
plt.show()

# Scatter plot: Income vs MntTotal

# Boxplot for Income to check outliers
